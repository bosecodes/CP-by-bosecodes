Q1. Convert a given number of days into months and days.
--------------------------------------------------------------------

#include<stdio.h> 
int main()
{
	int d, m, rd;
	printf("Enter no. of days: ");
	scanf("%d", &d);
	m=d/30;
	rd=d%30;
	printf("%d months and %d days.", m, rd);
	return 0;
}

Q17. Generate the following Pyramid Pattern:

			a 
			ab 
			abc 
			abcd
--------------------------------------------------------------------

#include<stdio.h>
int main()    
{     
	int c=97, i, j;
	for(i=1;i<=4;i++)
	{
		for(j=1;j<=i;j++)
		{
			printf("%c", (char)c);
			c++;
		}
		c=97;
		printf("\n");
	}
	return 0;  
}

Q18. Sort a series of numbers in either ascending order 
using Bubble Sort.
--------------------------------------------------------------------

#include<stdio.h>
int main()
{
	int n, i, j, t=0;
	printf("Enter no. of elements: ");
	scanf("%d", &n);
	int a[n];
	for(i=0;i<n;i++)
	{
		scanf("%d", &a[i]);
	}
	for(i=0;i<n-1;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]>a[j])
			{
				t=a[i];
				a[i]=a[j];
				a[j]=t;
			}
		}
	}
	for(i=0;i<n;i++)
	{
		printf("%d ", a[i]);
	}
}

Q18. Sort a series of numbers in either ascending order 
using Insertion Sort.
--------------------------------------------------------------------

#include <math.h> 
#include <stdio.h>
void insertionSort(int arr[], int n) 
{ 
	int i, key, j; 
	for (i = 1; i < n; i++) { 
		key = arr[i]; 
		j = i - 1; 
		while (j >= 0 && arr[j] > key) { 
			arr[j + 1] = arr[j]; 
			j = j - 1; 
		} 
		arr[j + 1] = key; 
	} 
} 
void printArray(int arr[], int n) 
{ 
	int i; 
	for (i = 0; i < n; i++) 
		printf("%d ", arr[i]); 
	printf("\n"); 
} 
int main() 
{ 
	int n, i;
	printf("Enter no. of elements: ");
	scanf("%d", &n);
	int arr[n];
	for(i=0;i<n;i++)
	{
		scanf("%d", &arr[i]);
	}
	insertionSort(arr, n); 
	printArray(arr, n); 
	return 0; 
}
